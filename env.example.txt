# Database Configuration for GeneVeda Biosciences
# Copy this content to a new file named .env.local in your project root

# MySQL Database Connection Settings
DB_HOST=localhost
DB_USER=root
DB_PASSWORD=your_mysql_password_here
DB_NAME=geneveda_biosciences
DB_PORT=3306

# Optional: SSL Configuration (for remote hosting)
# Set to "true" if your hosting requires SSL connection
DB_SSL=false

# Notes:
# - For local development, DB_HOST should be "localhost"
# - For remote hosting (like Hostinger), DB_HOST is usually "localhost" when app runs on same server
# - For remote MySQL, check your hosting panel for the correct hostname
# - DB_NAME is the database name you want to use (will be created automatically)
