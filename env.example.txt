# Database Configuration for GeneVeda Biosciences
# Copy this content to a new file named .env.local in your project root

# Prisma Database URL (Required for Prisma)
# Format: mysql://user:password@host:port/database
# Example for Hostinger:
# DATABASE_URL="mysql://u896634865_manprit:mnprt@Bharatail@localhost:3306/u896634865_manprit"
DATABASE_URL="mysql://user:password@host:port/database"

# Legacy MySQL Connection Settings (Optional - for backward compatibility)
# These are automatically converted to DATABASE_URL if DATABASE_URL is not set
DB_HOST=localhost
DB_USER=root
DB_PASSWORD=your_mysql_password_here
DB_NAME=geneveda_biosciences
DB_PORT=3306
DB_SSL=false

# Cloudinary Configuration (for image uploads)
CLOUDINARY_CLOUD_NAME=your_cloud_name
CLOUDINARY_API_KEY=your_api_key
CLOUDINARY_API_SECRET=your_api_secret

# Notes:
# - For Prisma, use DATABASE_URL format above
# - For local development, DB_HOST should be "localhost"
# - For remote hosting (like Hostinger), use the MySQL hostname provided by Hostinger
# - For Hostinger: DATABASE_URL="mysql://u896634865_manprit:password@mysql.hostinger.com:3306/u896634865_manprit"
# - URL encode special characters in password (e.g., @ becomes %40)
# - Get Cloudinary credentials from https://cloudinary.com/console
# - All images will be uploaded to "geneveda" folder in Cloudinary
